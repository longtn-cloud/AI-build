# Search (Plan 2 of 4) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a dedicated Search page where a user queries their own uploaded documents and gets back ranked raw passages — no LLM generation, pure retrieval over the embeddings Foundation already produces.

**Architecture:** A single new backend endpoint (`GET /search`) embeds the query via the existing Voyage AI `embed_query` service and runs a pgvector cosine-similarity query over `chunks` joined to `documents`, filtered to the authenticated user. The frontend adds a `/search` page and a small shared nav so users can move between Documents and Search.

**Tech Stack:** Same as Foundation — FastAPI / psycopg3 / pytest (backend); React 18 / Vite / TypeScript / React Router / Vitest + Testing Library (frontend); pgvector for similarity search.

## Global Constraints

- This is Plan 2 of 4 (Search). Chat Q&A and Quiz are out of scope — later plans build on what this one produces.
- No new tables. Reuses Foundation's `documents`/`chunks` tables and `embed_query`/`embed_texts` (Voyage AI, `voyage-3-lite`, 512-dim vectors).
- The only real per-user data isolation is the explicit `WHERE ... user_id = %s` clause the backend includes in every query — `chunks` has no `user_id` column itself, so the filter must go through a join to `documents`. This is not optional; every query touching `chunks`/`documents` must include it.
- Location granularity: results show `chunk_index` (1-based, "passage N of M") — there is no page-number tracking in this codebase, and this plan does not add any.
- Result scope: always return the top 10 chunks by cosine similarity, ranked best-first, with no minimum-similarity threshold. A user with zero ready documents/chunks gets an empty `results` array.
- Pure retrieval only — no LLM call anywhere in this plan.

Create `backend/app/routers/search.py`:

```python
from fastapi import APIRouter, Depends, HTTPException

from app.auth import get_current_user_id
from app.db import get_conn
from app.services.embeddings import embed_query

router = APIRouter(tags=["search"])


@router.get("/search")
async def search(q: str = "", user_id: str = Depends(get_current_user_id)):
    if not q.strip():
        raise HTTPException(status_code=400, detail="Query must not be empty")

    query_embedding = embed_query(q)

    with get_conn() as conn:
        rows = conn.execute(
            """
            SELECT
                d.id AS document_id,
                d.filename,
                c.chunk_index,
                c.content,
                1 - (c.embedding <=> %s::vector) AS score,
                count(*) OVER (PARTITION BY c.document_id) AS total_chunks
            FROM chunks c
            JOIN documents d ON d.id = c.document_id
            WHERE d.user_id = %s
            ORDER BY c.embedding <=> %s::vector
            LIMIT 10
            """,
            (query_embedding, user_id, query_embedding),
        ).fetchall()

    return {
        "results": [
            {
                "document_id": str(row["document_id"]),
                "filename": row["filename"],
                "chunk_index": row["chunk_index"],
                "total_chunks": row["total_chunks"],
                "content": row["content"],
                "score": row["score"],
            }
            for row in rows
        ]
    }
```

export type SearchResult = {
  document_id: string
  filename: string
  chunk_index: number
  total_chunks: number
  content: string
  score: number
}

export async function search(query: string): Promise<SearchResult[]> {
  const res = await fetch(`${API_BASE}/search?q=${encodeURIComponent(query)}`, {
    headers: await authHeader(),
  })
  if (!res.ok) throw new Error('Search failed')
  const data = await res.json()
  return data.results
}

```tsx
import { FormEvent, useState } from 'react'

import { search, SearchResult } from '../lib/api'

export function SearchPage() {
  const [query, setQuery] = useState('')
  const [results, setResults] = useState<SearchResult[] | null>(null)
  const [error, setError] = useState<string | null>(null)
  const [loading, setLoading] = useState(false)

  async function handleSubmit(event: FormEvent) {
    event.preventDefault()
    if (!query.trim()) return
    setLoading(true)
    setError(null)
    try {
      const found = await search(query)
      setResults(found)
    } catch {
      setError('Search failed, try again')
    } finally {
      setLoading(false)
    }
  }

  return (
    <div>
      <h1>Search</h1>
      <form onSubmit={handleSubmit}>
        <label htmlFor="search-input">Search your documents</label>
        <input id="search-input" value={query} onChange={(e) => setQuery(e.target.value)} />
        <button type="submit">Search</button>
      </form>
      {error && <p role="alert">{error}</p>}
      {loading && <p>Searching...</p>}
      {results !== null && !loading && results.length === 0 && <p>No results found</p>}
      {results !== null && !loading && results.length > 0 && (
        <ul>
          {results.map((r) => (
            <li key={`${r.document_id}-${r.chunk_index}`}>
              <span>
                {r.filename} — passage {r.chunk_index + 1} of {r.total_chunks}
              </span>
              <p>{r.content}</p>
            </li>
          ))}
        </ul>
      )}
    </div>
  )
}
```
